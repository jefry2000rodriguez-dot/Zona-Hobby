            loginForm.style.display = 'none';

            title.textContent = 'Crear Cuenta';

            registerForm.style.display = 'none';

    // Manejo de formularios
    handleLogin(e) {
        e.preventDefault();
        const email = document.getElementById('loginEmail').value;
        const password = document.getElementById('loginPassword').value;
        
        // Buscar usuario
        const user = this.users.find(u => u.email === email);
        
        if (!user) {
            this.showError('Usuario no encontrado');
            return;
        }
        
        if (user.isBanned) {
            this.showError('Usuario bloqueado por el administrador');
            return;
        }
        
        // En una aplicaci√≥n real, verificar contrase√±a
        this.currentUser = user;
        this.isAdmin = user.isAdmin;
        
        // Guardar en localStorage
        localStorage.setItem('zonaHobbyUser', JSON.stringify(user));
        
        this.updateUIForLoggedUser();
        this.closeAuthModal();
        this.showSuccess('¬°Bienvenido de vuelta!');
    }
    
    handleRegister(e) {
        e.preventDefault();
        const name = document.getElementById('registerName').value;
        const email = document.getElementById('registerEmail').value;
        const password = document.getElementById('registerPassword').value;
        const confirmPassword = document.getElementById('registerConfirmPassword').value;
        
        if (password !== confirmPassword) {
            this.showError('Las contrase√±as no coinciden');
            return;
        }
        
        if (this.users.find(u => u.email === email)) {
            this.showError('Este email ya est√° registrado');
            return;
        }
        
        // Crear nuevo usuario
        const newUser = {
            id: Date.now().toString(),
            name,
            email,
            isAdmin: false,
            isBanned: false
        };
        
        this.users.push(newUser);
        
        this.currentUser = newUser;
        this.isAdmin = false;
        
        localStorage.setItem('zonaHobbyUser', JSON.stringify(newUser));
        
        this.updateUIForLoggedUser();
        this.closeAuthModal();
        this.showSuccess('¬°Cuenta creada exitosamente!');
    }
    
    handleLogout() {
        this.currentUser = null;
        this.isAdmin = false;
        localStorage.removeItem('zonaHobbyUser');
        this.updateUIForLoggedUser();
        this.showSuccess('Sesi√≥n cerrada');
    }
    
    // Subida de fotos
    handleFileSelect(e) {
        const file = e.target.files[0];
        const display = document.getElementById('fileDisplay');
        
        if (file) {
            display.innerHTML = `
                <i data-lucide="check"></i>
                <span>${file.name}</span>
            `;

                <i data-lucide="upload"></i>
                <span>Haz clic para seleccionar una imagen</span>

        this.initIcons();

    handleUpload(e) {
        e.preventDefault();
        
        const title = document.getElementById('photoTitle').value;
        const description = document.getElementById('photoDescription').value;
        const category = document.getElementById('photoCategory').value;
        const tags = document.getElementById('photoTags').value;
        const file = document.getElementById('photoFile').files[0];
        
        if (!file) {
            this.showError('Por favor selecciona una imagen');
            return;
        }
        
        // Leer archivo como base64
        const reader = new FileReader();
        reader.onload = (event) => {
            const newPost = {
                id: Date.now().toString(),
                title,
                description,
                image: event.target.result,
                category,
                tags: tags ? tags.split(',').map(tag => tag.trim()) : [],
                author: this.currentUser.id,
                date: new Date().toISOString().split('T')[0],
                likes: [],
                comments: []
            };
            
            this.posts.unshift(newPost);
            this.renderFeed();
            this.closeUploadModal();
            this.showSuccess('¬°Foto publicada exitosamente!');
        };
        
        reader.readAsDataURL(file);
    }
    
    // Manejo del feed
    handleFilter(e) {
        const filter = e.target.dataset.category;
        this.currentFilter = filter;
        
        // Update active button
        document.querySelectorAll('.filter-btn').forEach(btn => {
            btn.classList.remove('active');
        });
        e.target.classList.add('active');
        
        this.currentPage = 0;
        this.renderFeed();
    }
    
    renderFeed() {
        const feedGrid = document.getElementById('feedGrid');
        let filteredPosts = this.posts;
        
        // Apply filter
        if (this.currentFilter !== 'todos') {
            filteredPosts = this.posts.filter(post => post.category === this.currentFilter);
        }
        
        // Pagination
        const startIndex = this.currentPage * this.postsPerPage;
        const endIndex = startIndex + this.postsPerPage;
        const postsToShow = filteredPosts.slice(startIndex, endIndex);
        
        feedGrid.innerHTML = postsToShow.map(post => this.createPostCard(post)).join('');
        
        // Hide load more button if no more posts
        const loadMoreBtn = document.getElementById('loadMoreBtn');
        if (endIndex >= filteredPosts.length) {
            loadMoreBtn.style.display = 'none';
        } else {
            loadMoreBtn.style.display = 'block';
        }
        
        this.initIcons();
    }
    
    createPostCard(post) {
        const author = this.users.find(u => u.id === post.author);
        const isLiked = this.currentUser && post.likes.includes(this.currentUser.id);
        
        return `
            <div class="post-card" onclick="zonaHobby.openPhotoDetail('${post.id}')">
                <img src="${post.image}" alt="${post.title}" class="post-image">
                <div class="post-content">
                    <div class="post-header">
                        <h3 class="post-title">${post.title}</h3>
                        <span class="post-category">${this.getCategoryLabel(post.category)}</span>
                    </div>
                    <p class="post-description">${post.description}</p>
                    <div class="post-meta">
                        <span class="post-author">por ${author ? author.name : 'Usuario'}</span>
                        <span class="post-date">${this.formatDate(post.date)}</span>
                    </div>
                    <div class="post-actions">
                        <button class="btn-like ${isLiked ? 'liked' : ''}" onclick="zonaHobby.toggleLike('${post.id}', event)">
                            <i data-lucide="heart"></i>
                            <span>${post.likes.length}</span>
                        </button>
                        <button class="btn-admin-action admin-only" onclick="zonaHobby.deletePost('${post.id}', event)">
                            <i data-lucide="trash-2"></i>
                            Eliminar
                        </button>
                    </div>
                    <div class="post-tags">
                        ${post.tags.map(tag => `<span class="tag">#${tag}</span>`).join('')}
                    </div>
                </div>
            </div>
        `;
    }
    
    loadMorePosts() {
        this.currentPage++;
        this.renderFeed();
    }
    
    // Detalles de foto
    openPhotoDetail(postId) {
        const post = this.posts.find(p => p.id === postId);
        if (!post) return;
        
        this.currentPhotoDetail = post;
        
        document.getElementById('detailPhotoImage').src = post.image;
        document.getElementById('detailPhotoTitle').textContent = post.title;
        document.getElementById('detailPhotoDescription').textContent = post.description;
        document.getElementById('detailPhotoCategory').textContent = this.getCategoryLabel(post.category);
        document.getElementById('detailPhotoDate').textContent = this.formatDate(post.date);

        tagsContainer.innerHTML = post.tags.map(tag => `<span class="tag">#${tag}</span>`).join('');
        
        const likeBtn = document.getElementById('detailLikeBtn');
        const likeCount = document.getElementById('detailLikeCount');
        const isLiked = this.currentUser && post.likes.includes(this.currentUser.id);
        
        likeBtn.className = `btn-like ${isLiked ? 'liked' : ''}`;
        likeCount.textContent = post.likes.length;

        const deleteBtn = document.getElementById('detailDeleteBtn');
        if (this.isAdmin) {
            deleteBtn.style.display = 'block';
        } else {
            deleteBtn.style.display = 'none';
        }
        
        // Render comments
        this.renderComments(post.comments);
        
        const modal = document.getElementById('photoDetailModal');

    closePhotoDetailModal() {
        const modal = document.getElementById('photoDetailModal');
        modal.classList.remove('active');
        document.body.style.overflow = 'auto';
        this.currentPhotoDetail = null;
    }
    
    renderComments(comments) {
        const commentsList = document.getElementById('commentsList');
        
        if (comments.length === 0) {
            commentsList.innerHTML = '<p style="color: var(--neutral-500); text-align: center;">No hay comentarios a√∫n</p>';

        }
        
        commentsList.innerHTML = comments.map(comment => {
            const author = this.users.find(u => u.id === comment.author);
            return `
                <div class="comment">
                    <div class="comment-author">${author ? author.name : 'Usuario'}</div>
                    <div class="comment-text">${comment.text}</div>
                    <div class="comment-date">${this.formatDateTime(comment.date)}</div>
                </div>
            `;
        }).join('');
    }
    
    handleComment(e) {
        e.preventDefault();
        
        if (!this.currentUser) {
            this.showError('Debes iniciar sesi√≥n para comentar');
            return;
        }
        
        const input = document.getElementById('commentInput');
        const text = input.value.trim();
        
        if (!text) return;
        
        const newComment = {
            id: Date.now().toString(),
            author: this.currentUser.id,
            text,
            date: new Date().toISOString()
        };
        
        this.currentPhotoDetail.comments.push(newComment);
        
        input.value = '';
        this.renderComments(this.currentPhotoDetail.comments);
        this.showSuccess('Comentario agregado');
    }
    
    // Likes
    toggleLike(postId, event) {
        event.stopPropagation();
        
        if (!this.currentUser) {
            this.showError('Debes iniciar sesi√≥n para dar like');
            return;
        }
        
        const post = this.posts.find(p => p.id === postId);
        if (!post) return;
        
        const likeIndex = post.likes.indexOf(this.currentUser.id);
        
        if (likeIndex > -1) {
            post.likes.splice(likeIndex, 1);
        } else {
            post.likes.push(this.currentUser.id);
        }
        
        this.renderFeed();
        
        if (this.currentPhotoDetail && this.currentPhotoDetail.id === postId) {
            this.openPhotoDetail(postId);
        }
    }
    
    handleDetailLike() {
        if (this.currentPhotoDetail) {
            this.toggleLike(this.currentPhotoDetail.id, { stopPropagation: () => {} });
        }
    }
    
    // Admin functions
    deletePost(postId, event) {
        event.stopPropagation();
        
        if (!this.isAdmin) {
            this.showError('No tienes permisos de administrador');
            return;
        }
        
        if (confirm('¬øEst√°s seguro de que quieres eliminar esta publicaci√≥n?')) {
            this.posts = this.posts.filter(p => p.id !== postId);
            this.renderFeed();
            
            if (this.currentPhotoDetail && this.currentPhotoDetail.id === postId) {
                this.closePhotoDetailModal();
            }
            
            this.showSuccess('Publicaci√≥n eliminada');
        }
    }
    
    handleDetailDelete() {
        if (this.currentPhotoDetail) {
            this.deletePost(this.currentPhotoDetail.id, { stopPropagation: () => {} });
        }
    }

    renderStats() {
        const totalUsers = this.users.length;
        const totalPosts = this.posts.length;
        const totalComments = this.posts.reduce((sum, post) => sum + post.comments.length, 0);
        const totalLikes = this.posts.reduce((sum, post) => sum + post.likes.length, 0);
        
        document.getElementById('totalUsers').textContent = totalUsers;
        document.getElementById('totalPosts').textContent = totalPosts;
        document.getElementById('totalComments').textContent = totalComments;
        document.getElementById('totalLikes').textContent = totalLikes;
    }
    
    toggleUserBan(userId) {
        if (!this.isAdmin) {
            this.showError('No tienes permisos de administrador');
            return;
        }
        
        const user = this.users.find(u => u.id === userId);
        if (!user) return;
        
        if (user.id === this.currentUser?.id) {
            this.showError('No puedes bloquear tu propia cuenta');
            return;
        }
        
        user.isBanned = !user.isBanned;
        this.renderUsersList();
        
        const action = user.isBanned ? 'bloqueado' : 'desbloqueado';
        this.showSuccess(`Usuario ${action} exitosamente`);
    }

    getUserName(userId) {
        const user = this.users.find(u => u.id === userId);

            'familia': 'Familia',
            'viajes': 'Viajes',
            'momentos': 'Momentos'
        };
        return labels[category] || category;
    }
    
    formatDate(dateString) {
        const date = new Date(dateString);
        return date.toLocaleDateString('es-ES', {
            year: 'numeric',
            month: 'long',
            day: 'numeric'
        });
    }
    
    formatDateTime(dateString) {
        const date = new Date(dateString);
        return date.toLocaleDateString('es-ES', {
            year: 'numeric',
            month: 'short',
            day: 'numeric',
            hour: '2-digit',
            minute: '2-digit'
        });
    }
    
    showError(message) {
        this.showNotification(message, 'error');
    }
    
    showSuccess(message) {
        this.showNotification(message, 'success');
    }
    
    showNotification(message, type = 'info') {
        // Create notification element
        const notification = document.createElement('div');
        notification.className = `notification ${type}`;
        notification.textContent = message;
        
        // Add styles
        notification.style.cssText = `
            position: fixed;
            top: 20px;
            right: 20px;
            background: ${type === 'error' ? 'var(--error)' : type === 'success' ? 'var(--success)' : 'var(--primary-500)'};
            color: white;
            padding: var(--space-sm) var(--space-md);
            border-radius: var(--radius-md);
            box-shadow: var(--shadow-lg);
            z-index: 3000;
            font-weight: 600;
            animation: slideInRight 0.3s ease-out;
        `;
        
        document.body.appendChild(notification);
        
        // Remove after 3 seconds
        setTimeout(() => {
            notification.style.animation = 'slideOutRight 0.3s ease-in';
            setTimeout(() => {
                document.body.removeChild(notification);
            }, 300);
        }, 3000);
    }
    
    handleOutsideClick(e) {
        // Close dropdowns when clicking outside
        const userDropdown = document.getElementById('userDropdown');
        const userMenu = document.getElementById('userMenu');
        
        if (!userMenu.contains(e.target)) {
            userDropdown.classList.remove('active');
        }
    }
    
    handleKeyboard(e) {
        // Close modals with Escape key
        if (e.key === 'Escape') {
            const modals = document.querySelectorAll('.modal.active');
            if (modals.length > 0) {
                const modal = modals[modals.length - 1];
                if (modal.id === 'photoDetailModal') {
                    this.closePhotoDetailModal();
                } else if (modal.id === 'uploadModal') {
                    this.closeUploadModal();
                } else if (modal.id === 'authModal') {
                    this.closeAuthModal();
                }
            }
        }
    }
    
    // Initialize icons
    initIcons() {
        if (typeof lucide !== 'undefined') {
            lucide.createIcons();
        }
    }
}
